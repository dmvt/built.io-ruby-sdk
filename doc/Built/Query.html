def exec
  if !@class_uid
    raise BuiltError, I18n.t("querying.class_uid")
  end

  uri = Built::Object.uri(@class_uid)

  QueryResponse.new(
    Built.client.request(uri, :get, nil, @params).parsed_response,
    @class_uid
  )
end
